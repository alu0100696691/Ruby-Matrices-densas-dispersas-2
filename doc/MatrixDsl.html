def operand(fil1,fil2)
         n = fil1.size
         m = fil2.size
         case @typeMatrix
         when "dense" 
                 @DM << SparseMatrix::DenseMatrix.new(n,m,[fil1,fil2])
         when "sparse" 
                 @DM << SparseMatrix::SparseMatrix.new(n,m,[fil1,fil2])
         end 
end

def option(opt)
         case opt
         when "dense"  
                 @typeMatrix = "dense"
         when "sparse" 
                 @typeMatrix = "sparse"
         when "console"
                 @represent = 1
         when "matrix"
                 @represent = 0
         end               
end

def result(res)
         case @name
         when "suma"
                 printf "\t\tSUMA DE MATRICES \n"
                 printf "\t\t----------------\n\n"
                 printf " OPERANDO A:\t%s\n ", @DM[0].to_s
                 printf "OPERANDO B:\t%s\n\n ", @DM[1].to_s
                 printf "RESULTADO:\t%s\n\n\n", res
         when "multiplicar"                                  
                 printf "\t\tMULTIPLICACION DE MATRICES \n"
                 printf "\t\t--------------------------\n\n"
                 printf " OPERANDO A:\t%s\n ", @DM[0].to_s

                 printf "RESULTADO:\t%s\n\n\n", res

 end

    # File lib/sparse_matrix.rb, line 471
def run
         case @name
         when "suma"
                 resultado = (@DM[0]+@DM[1]).to_s
         when "multiplicar"
                 resultado = (@DM[0]*@DM[1]).to_s
         end

         if @represent == 1
                 result(resultado)
         else
                 return resultado
         end                  
end
